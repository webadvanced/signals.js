  8 /**
  9 * @namespace Signals Namespace - A lightweight (1k minified) pure JavaScript PubSub library
 10 * @name signals
 11 */
 12 var signals = ( function ( global, undefined ) {
 13     "use strict";
 14 	var evts = [],
 15 	makeObservable,
 16 	subscribeToSignal,
 17 	takeAction,
 18 	getSignalType,
 19 	fire,
 20 	unload,
 21     iterator,
 22     FUNCTION = 'function',
 23     checkArgument,
 24     createProxyFor;
 25 	/**
 26 	* Private function that checks a given argument agents is expected type.
 27 	* <br />- <strong>(will throw is arg does not match expected)</strong>
 28 	* @private
 29 	* @name checkArgument
 30 	* @param {object} [arg] Argument.
 31 	* @param {object} [expected] Expected type.
 32 	* @param {string} [message] Message to throw. (default = 'Invalid argument').
 33 	*/
 34 	checkArgument = function( arg, expected, message ) {
 35 		if ( typeof arg !== expected ) {
 36 			throw message || 'Invalid argument.';
 37 		}
 38 	};
 39 	/**
 40 	* Private function that makes the given signal observable.
 41 	* <br />- <strong>(signals are an arbitrary string)</strong>
 42 	* @private
 43 	* @name makeObservable
 44 	* @param {string} [signal] Name of the signal. (default = 'any').
 45 	*/
 46 	makeObservable = function ( signal ) {
 47 		var signalType = getSignalType( signal );
 48 		if ( evts[signalType] === undefined ) {
 49 			evts[signalType] = [];
 50 		}
 51 	};
 52 	/**
 53 	* Private function that subscribes a listener to a signal.
 54 	* <br />- <strong>(signals are an arbitrary string)</strong>
 55 	* @private
 56 	* @name subscribeToSignal
 57 	* @param {string} [signalName] Name of the signal. (default = 'any').
 58 	* @param {Object} [callbacks] Callback function or Array of functions to be bound to the signal.
 59 	* @param {Object} [context] Context on which listener will be executed (object that should represent the `this`. (default = window)
 60 	*/
 61     subscribeToSignal = function ( signalName, callbacks, context ) {
 62 		var signalType = getSignalType( signalName ), 
 63 			i = 0, 
 64 			l,
 65 			callback;
 66 		if ( callbacks instanceof Array ) {
 67 			l = callbacks.length;
 68 			for (i; i < l; i++) {
 69 				callback = callbacks[i];
 70 				checkArgument( callback, FUNCTION, 'Callback must be a function' );	
 71 				evts[signalType].push( {callback: callback, obj: context || global} );
 72 			}
 73 		} else {
 74 			checkArgument(callbacks, FUNCTION, 'Callback must be a function');
 75 			evts[signalType].push({callback: callbacks, obj: context || global});
 76 		}
 77     };
 78     /**
 79     * Private function that returns a function that will handle the execution of Listeners for a given Signal
 80     * @private
 81     * @name createProxyFor
 82     * @param {string} [signalName] Name of the signal. (default = 'any').
 83     */
 84     createProxyFor = function ( signalName ) {
 85         return (function( arg ) {
 86             takeAction( 'broadcast', signalName, arg );
 87         });
 88     };
 89 	/**
 90 	* Private function that handles the execution of Listeners for a given Signal or unsubscribe the Listener from the Signal
 91 	* @private
 92 	* @name takeAction
 93 	* @param {string} [action] The action the function should take with the given arguments.
 94 	* @param {string} [signalName] Name of the signal. (default = 'any').
 95 	* @param {Object} [arg] The args that will be passed to the Listeners or the function to be unsubscribed.
 96 	*/
 97     takeAction = function ( action, signalName, arg ) {
 98         var signalType = getSignalType( signalName ),
 99             actionsBefore = evts[signalType + ':before'],
100             actions = evts[signalType],
101             actionsAfter = evts[signalType + ':after'],
102 			takeAction;
103 
104         if ( actions === undefined ) {
105             return;
106         }
107 		if ( action === 'broadcast' ) {
108 			takeAction = fire;
109 		} else {
110 			takeAction = unload;
111 		}
112         if( actionsBefore ) {
113             iterator( actionsBefore, takeAction, arg );
114         }
115 
116         iterator( actions, takeAction, arg );
117 
118         if( actionsAfter ) {
119             iterator( actionsAfter, takeAction, arg );
120         }
121     };
122     iterator = function( collection, fn, arg ) {
123         var i = 0, l = collection.length;
124         for ( ; i < l; i += 1 ) {
125             fn( collection[i], arg, collection, i );
126         }
127     };
128 	/**
129 	* Private function that fires off a Listener.
130 	* @private
131 	* @name fire
132 	* @param {Object} [listener] The object that holds a callback and executing context.
133 	* @param {Object} [arg] The args that will be passed to the Listener.
134 	*/
135 	fire = function ( listener, arg ) {
136 		var ctx = listener.obj,
137 			fn = listener.callback;
138 		global.setTimeout( function () { fn.call(ctx, arg); }, 0 );
139 	};
140 	/**
141 	* Private function that will unsubscribe a function from the supplied Signal.
142 	* @private
143 	* @name unload
144 	* @param {Object} [listener] The object that holds a callback and executing context.
145 	* @param {Function} [arg] The Function to be removed from the Signal.
146 	* @param {Array} [actions] Collection of subscribers for a signal.
147 	* @param {Number} [i] Index in the actions collection to be spliced out of the array.
148 	*/
149 	unload = function ( listener, arg, actions, i ) {
150 		if ( listener !== undefined && listener.callback === arg ) {
151 			actions.splice( i, 1 );
152         }
153 	};
154 	/**
155 	* Private function that sets the signal.
156 	* @private
157 	* @name getSignalType
158 	* @param {string} [signal] The Signal.
159 	*/
160     getSignalType = function ( signal ) {
161         /**
162 		* If signal is undefined, set signal to 'any'.
163 		* @signal string
164 		* @private
165 		*/
166 		return signal || 'any';
167     };
168 	
169     return {
170 		/**
171 		* Add a Listener or collection of listeners to the given Signal.
172 		* @name signals.subscribe
173 		* @param {string} The signals name.
174 		* @param {Object}  Array of Functions or function that should be added to the Signal.
175 		* @param {Object} [context] Context on which the listener will be executed (object that should represent the `this`. (default = window)
176 		* @function
177 		*/
178         subscribe: function ( signalName, callbacks, context ) {
179             makeObservable( signalName );
180             subscribeToSignal( signalName, callbacks, context );
181         },
182 		/**
183 		* Broadcast to all listeners of the given signal to execute.
184 		* @name signals.broadcast
185 		* @param {string} [signalName] The signal.
186 		* @param {Object} [arg] Will be passed to each function as the argument.
187 		* @function
188 		*/
189         broadcast: function ( signalName, arg ) {
190             takeAction( 'broadcast', signalName, arg );
191         },
192         /**
193         * Return a broadcast function for the given signal.
194         * @name signals.proxy
195         * @param {string} [signalName] The signal.
196         * @return {function} when executed will broadcast for given sigalName.
197         * @function
198         */
199         proxy: function ( signalName ) {
200             return createProxyFor( signalName );
201         },
202 		/**
203 		* Remove a listener from the given signal.
204 		* @name signals.unsubscribe
205 		* @param {string} [signalName] The signal.
206 		* @param {Function} Function that should be removed.
207 		* @function
208 		*/
209         unsubscribe: function ( signalName, fn ) {
210             takeAction( 'unsubscribe', signalName, fn );
211         },
212 		/**
213 		* Check if Signal is observable.
214 		* @name signals.isObservable
215 		* @param {string} [signalName] The signal.
216 		* @return {boolean} if signal is observable.
217 		* @function
218 		*/
219         isObservable: function ( signalName ) {
220             var signalType = getSignalType( signalName );
221             return evts[signalType] !== undefined;
222         },
223 		/**
224 		* Gets the count of Listeners for the given Signal.
225 		* @name signals.listenerCount
226 		* @param {string} [signalName] The signal.
227 		* @return {Number} Number of Listeners. (default = 0).
228 		* @function
229 		*/
230         listenerCount: function ( signalName ) {
231             var signalType = getSignalType( signalName ),
232                 actions = evts[signalType];
233             return actions !== undefined ? actions.length : 0;
234         }
235     };
236 }(this));
