  8 /**
  9 * @namespace Signals Namespace - A lightweight (1k minified) pure JavaScript PubSub library
 10 * @name signals
 11 */
 12 var signals = ( function ( global, undefined ) {
 13     "use strict";
 14     var evts = [],
 15     makeObservable,
 16     subscribeToSignal,
 17     takeAction,
 18     getSignalType,
 19     fire,
 20     unload,
 21     iterator,
 22     FUNCTION = 'function',
 23     checkArgument,
 24     createProxyFor,
 25     ActionScope;
 26 
 27     /**
 28     * Private function that checks a given argument agents is expected type.
 29     * <br />- <strong>(will throw is arg does not match expected)</strong>
 30     * @private
 31     * @name checkArgument
 32     * @param {object} [arg] Argument.
 33     * @param {object} [expected] Expected type.
 34     * @param {string} [message] Message to throw. (default = 'Invalid argument').
 35     */
 36     checkArgument = function( arg, expected, message ) {
 37         if ( typeof arg !== expected ) {
 38             throw message || 'Invalid argument.';
 39         }
 40     };
 41     /**
 42     * Private function that makes the given signal observable.
 43     * <br />- <strong>(signals are an arbitrary string)</strong>
 44     * @private
 45     * @name makeObservable
 46     * @param {string} [signal] Name of the signal. (default = 'any').
 47     */
 48     makeObservable = function ( signal ) {
 49         var signalType = getSignalType( signal );
 50         if ( evts[signalType] === undefined ) {
 51             evts[signalType] = [];
 52         }
 53     };
 54     /**
 55     * Private function that subscribes a listener to a signal.
 56     * <br />- <strong>(signals are an arbitrary string)</strong>
 57     * @private
 58     * @name subscribeToSignal
 59     * @param {string} [signalName] Name of the signal. (default = 'any').
 60     * @param {Object} [callbacks] Callback function or Array of functions to be bound to the signal.
 61     * @param {Object} [context] Context on which listener will be executed (object that should represent the `this`. (default = window)
 62     */
 63     subscribeToSignal = function ( signalName, callbacks, context ) {
 64         var signalType = getSignalType( signalName ), 
 65             i = 0, 
 66             l,
 67             callback;
 68         if ( callbacks instanceof Array ) {
 69             l = callbacks.length;
 70             for (i; i < l; i++) {
 71                 callback = callbacks[i];
 72                 checkArgument( callback, FUNCTION, 'Callback must be a function' );    
 73                 evts[signalType].push( {callback: callback, obj: context || global} );
 74             }
 75         } else {
 76             checkArgument(callbacks, FUNCTION, 'Callback must be a function');
 77             evts[signalType].push({callback: callbacks, obj: context || global});
 78         }
 79     };
 80     /**
 81     * Private function that returns a function that will handle the execution of Listeners for a given Signal
 82     * @private
 83     * @name createProxyFor
 84     * @param {string} [signalName] Name of the signal. (default = 'any').
 85     */
 86     createProxyFor = function ( signalName ) {
 87         return (function( arg ) {
 88             takeAction( 'broadcast', signalName, arg );
 89         });
 90     };
 91     /**
 92     * Private function that handles the execution of Listeners for a given Signal or unsubscribe the Listener from the Signal
 93     * @private
 94     * @name takeAction
 95     * @param {string} [action] The action the function should take with the given arguments.
 96     * @param {string} [signalName] Name of the signal. (default = 'any').
 97     * @param {Object} [args] The args that will be passed to the Listeners or the function to be unsubscribed.
 98     */
 99     takeAction = function ( action, signalName, args ) {
100         var signalType = getSignalType( signalName ),
101             actionsBefore = evts[signalType + ':before'],
102             actions = evts[signalType],
103             actionsAfter = evts[signalType + ':after'],
104             takeAction;
105 
106         if ( actions === undefined ) {
107             return;
108         }
109         
110         takeAction = ( action === 'broadcast' ) ? fire : unload; 
111         
112         ActionScope.init( actionsBefore, actions, actionsAfter ).run( takeAction, args );
113     };
114     
115     ActionScope = function( beforeActions, actions, afterActions ) {
116         var self = this;
117         self.processed = 0;
118         self.beforeActions = beforeActions;
119         self.actions = actions;
120         self.afterActions = afterActions;
121         self.processTotal = actions.length + ( ( beforeActions ) ? beforeActions.length : 0 );
122 
123         return self;
124     };
125 
126     ActionScope.init = function( beforeActions, actions, afterActions ) {
127         return new ActionScope( beforeActions, actions, afterActions );
128     };
129     
130     ActionScope.fn = ActionScope.prototype;
131     
132     ActionScope.fn.run = function( takeAction, args ) {
133         var self = this,
134             trackAction;
135         self.args = args;
136         self.takeAction = takeAction;
137         //keep scope
138         trackAction = function() {
139             self.track();
140         };
141 
142         if( self.beforeActions ) {
143             iterator( self.beforeActions, takeAction, args, trackAction );
144         }
145 
146         iterator( self.actions, takeAction, args, trackAction );
147     };
148     
149     ActionScope.fn.track = function() {
150         var self = this;
151         self.processed++;
152         
153         if( self.processTotal !== self.processed ) {
154             return self;
155         }
156 
157         if( self.afterActions ) {
158             iterator( self.afterActions, self.takeAction, self.args, function() {} );
159         }
160     };
161 
162     iterator = function( collection, fn, args, trackAction, i ) {
163         i = i || 0;
164         //do work
165         fn( collection[i], args, collection, i );
166         trackAction();
167         //incrament count
168         i++;
169         //check count < collection
170         if( i < collection.length ) {
171             //call iterator again
172             global.setTimeout( function() {
173                 iterator( collection, fn, args, trackAction, i );
174             }, 10);
175         }
176     };
177     /**
178     * Private function that fires off a Listener.
179     * @private
180     * @name fire
181     * @param {Object} [listener] The object that holds a callback and executing context.
182     * @param {Object} [arg] The args that will be passed to the Listener.
183     */
184     fire = function ( listener, arg ) {
185         var ctx = listener.obj,
186             fn = listener.callback;
187         fn.call(ctx, arg);
188     };
189     /**
190     * Private function that will unsubscribe a function from the supplied Signal.
191     * @private
192     * @name unload
193     * @param {Object} [listener] The object that holds a callback and executing context.
194     * @param {Function} [arg] The Function to be removed from the Signal.
195     * @param {Array} [actions] Collection of subscribers for a signal.
196     * @param {Number} [i] Index in the actions collection to be spliced out of the array.
197     */
198     unload = function ( listener, arg, actions, i ) {
199         if ( listener !== undefined && listener.callback === arg ) {
200             actions.splice( i, 1 );
201         }
202     };
203     /**
204     * Private function that sets the signal.
205     * @private
206     * @name getSignalType
207     * @param {string} [signal] The Signal.
208     */
209     getSignalType = function ( signal ) {
210         /**
211         * If signal is undefined, set signal to 'any'.
212         * @signal string
213         * @private
214         */
215         return signal || 'any';
216     };
217     
218     return {
219         /**
220         * Add a Listener or collection of listeners to the given Signal.
221         * @name signals.subscribe
222         * @param {string} The signals name.
223         * @param {Object}  Array of Functions or function that should be added to the Signal.
224         * @param {Object} [context] Context on which the listener will be executed (object that should represent the `this`. (default = window)
225         * @function
226         */
227         subscribe: function ( signalName, callbacks, context ) {
228             makeObservable( signalName );
229             subscribeToSignal( signalName, callbacks, context );
230         },
231         /**
232         * Broadcast to all listeners of the given signal to execute.
233         * @name signals.broadcast
234         * @param {string} [signalName] The signal.
235         * @param {Object} [arg] Will be passed to each function as the argument.
236         * @function
237         */
238         broadcast: function ( signalName, arg ) {
239             takeAction( 'broadcast', signalName, arg );
240         },
241         /**
242         * Return a broadcast function for the given signal.
243         * @name signals.proxy
244         * @param {string} [signalName] The signal.
245         * @return {function} when executed will broadcast for given sigalName.
246         * @function
247         */
248         proxy: function ( signalName ) {
249             return createProxyFor( signalName );
250         },
251         /**
252         * Remove a listener from the given signal.
253         * @name signals.unsubscribe
254         * @param {string} [signalName] The signal.
255         * @param {Function} Function that should be removed.
256         * @function
257         */
258         unsubscribe: function ( signalName, fn ) {
259             takeAction( 'unsubscribe', signalName, fn );
260         },
261         /**
262         * Check if Signal is observable.
263         * @name signals.isObservable
264         * @param {string} [signalName] The signal.
265         * @return {boolean} if signal is observable.
266         * @function
267         */
268         isObservable: function ( signalName ) {
269             var signalType = getSignalType( signalName );
270             return evts[signalType] !== undefined;
271         },
272         /**
273         * Gets the count of Listeners for the given Signal.
274         * @name signals.listenerCount
275         * @param {string} [signalName] The signal.
276         * @return {Number} Number of Listeners. (default = 0).
277         * @function
278         */
279         listenerCount: function ( signalName ) {
280             var signalType = getSignalType( signalName ),
281                 actions = evts[signalType];
282             return actions !== undefined ? actions.length : 0;
283         }
284     };
285 }(this));
